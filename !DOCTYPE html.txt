        // 모달 열기 함수
        function openModal(cardElement) {
            const id = cardElement.dataset.articleId;
            const title = cardElement.querySelector('[data-title]').dataset.title;
            const tag = cardElement.querySelector('[data-tag]').dataset.tag;
            const source = cardElement.querySelector('[data-source]').dataset.source;
            const date = cardElement.querySelector('[data-date]').dataset.date;
            const imageUrl = cardElement.querySelector('img').dataset.imageUrl;

            const contentData = articleMockContents[id];

            if (contentData) {
                // 모달 내용 채우기
                modalTitle.textContent = title;
                modalTag.textContent = tag;
                // [수정된 부분] 템플릿 리터럴(\` \`) 앞의 역슬래시를 제거하여 SyntaxError 해결
                modalTag.className = `text-xs font-semibold px-3 py-1 text-white rounded-full mb-3 inline-block ${contentData.tagColor}`;
                modalImage.src = imageUrl;
                modalSource.textContent = source;
                modalDate.textContent = date;
                modalFullContent.innerHTML = contentData.fullContent;

                // 모달 표시
                modalOverlay.style.display = 'flex';
                document.body.style.overflow = 'hidden'; // 스크롤 잠금
            } else {
                console.error('Article content not found for ID:', id);
            }
        }

        // 모달 닫기 함수
        function closeModal() {
            modalOverlay.style.display = 'none';
            document.body.style.overflow = ''; // 스크롤 잠금 해제
        }

        // 이벤트 리스너 등록
        document.addEventListener('DOMContentLoaded', () => {
            // 1. 기사 카드 클릭 이벤트
            articleCards.forEach(card => {
                // <a> 태그 대신 <div>를 클릭하면 모달이 열리도록 처리
                card.addEventListener('click', (e) => {
                    e.preventDefault(); // 기본 링크 이동 방지
                    openModal(card);
                });
            });

            // 2. 모달 닫기 버튼 클릭 이벤트
            modalCloseBtn.addEventListener('click', closeModal);

            // 3. 모달 외부 클릭 이벤트 (오버레이 클릭 시 닫기)
            modalOverlay.addEventListener('click', (e) => {
                if (e.target === modalOverlay) {
                    closeModal();
                }
            });

            // 4. ESC 키 입력 시 닫기
            document.addEventListener('keydown', (e) => {
                if (e.key === 'Escape' && modalOverlay.style.display === 'flex') {
                    closeModal();
                }
            });
        });
